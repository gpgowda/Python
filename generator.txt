def count_up_to(max):
    num = 1
    while num <= max:
        yield num
        num += 1


# Using the generator
counter = count_up_to(3)

print(next(counter))
print(next(counter))
print(next(counter))


# Generator in a for loop
def even_numbers(limit):
    for i in range(limit):
        if i % 2 == 0:
            yield i


for num in even_numbers(10):
    print(num, end=" ")
